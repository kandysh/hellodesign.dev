Data Fetch Optimization

Analysis indicates an N+1 query issue in the user feed service. Abstracting the fetch logic to utilize batch processing and pre-compiled views will reduce latency by an estimated 42%.

Proposed Refactor
// Current Implementation (Bottleneck)
async function getUserFeed(userIds) {
  const feed = [];
  for (const id of userIds) {
    const data = await db.query(`SELECT * FROM posts WHERE user_id = ${id}`);
    feed.push(...data);
  }
  return feed;
}

// Proposed Refactor (Batched Query)
async function getUserFeedOptimized(userIds) {
  if (!userIds.length) return [];
  
  const placeholders = userIds.map((_, i) => `$${i + 1}`).join(',');
  const query = `
    SELECT * FROM posts 
    WHERE user_id IN (${placeholders})
    ORDER BY created_at DESC
  `;
  
  return await db.query(query, userIds);
}
